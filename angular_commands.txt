
g : generate
c : component
s : service


ng new employees-frontend   //crear proyecto angular
ng serve -o                 //ejecutar la aplicación
ng serve                    //compile app for development
ng build --prod             //(until version 14)complie app for production
ng build --configuration "production" //(until version 17)complie app for production
ng g class employee         //generate employee class
ng g c employees-list       //generate component employee-list
ng g c register-employee    //generate component register-employee
ng g s employee             //generate employee service
npm i --save sweetalert2@7.26.9 //instalar sweetalert2
